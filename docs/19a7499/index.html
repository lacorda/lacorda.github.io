cookie

官方文档

允许用户创建、获取、删除、查询cookie

"permissions": [
  "cookies"
]

方法​

增​

chrome.cookies.set( 
  details: object,
  callback?: function,
)
chrome.cookies.set({
  url: 'https://www.baidu.com',
  name: 'name',
  value: 'value',
  domain: 'baidu.com',
  path: '/',
  secure: true,
  httpOnly: true,
  expirationDate: Date.now() + 1000 * 60 * 60 * 24 * 365 * 10,
}, (cookie) => {
  console.log('🍄  background: >>>>>>>>>>>>>>>>>> 创建cookie', Date.now(), cookie);
});

删​

chrome.cookies.remove(
  details: object,
  callback?: function,
)
chrome.cookies.remove({
  url: 'https://www.baidu.com',
  name: 'name',
}, (cookie) => {
  console.log('🍄  background: >>>>>>>>>>>>>>>>>> 删除cookie', Date.now(), cookie);
});

改​

chrome.cookies.set(
  details: object,
  callback?: function,
)
chrome.cookies.set({
  url: 'https://www.baidu.com',
  name: 'name',
  value: 'value',
  domain: 'baidu.com',
  path: '/',
  secure: true,
  httpOnly: true,
  expirationDate: Date.now() + 1000 * 60 * 60 * 24 * 365 * 10,
}, (cookie) => {
  console.log('🍄  background: >>>>>>>>>>>>>>>>>> 创建cookie', Date.now(), cookie);
});

查​

chrome.cookies.get(
  details: object,
  callback?: function,
)
chrome.cookies.get({
  url: 'https://www.baidu.com',
  name: 'name',
}, (cookie) => {
  console.log('🍄  background: >>>>>>>>>>>>>>>>>> 获取cookie', Date.now(), cookie);
});

chrome.cookies.getAll({
  url: 'https://www.baidu.com',
}, (cookie) => {
  console.log('🍄  background: >>>>>>>>>>>>>>>>>> 获取所有cookie', Date.now(), cookie);
});
